SQL指令應用(查詢2))


1.使用SQL FN完成欄位-平均

select avg(`欄位名稱`) from `資料表名稱`;

2.合併欄位字串

SELECT concat(`欄位名稱`,' ','先生/小姐') FROM `資料表名稱`;

3.從右邊抓取指定數文字

SELECT right(`欄位名稱`,3) FROM `資料表名稱`;

4.從第2字元取3個值

SELECT substring(`欄位名稱`,2,3) FROM `資料表名稱`;

5.指定特定多重值篩選

SELECT * FROM `資料表名稱` WHERE `欄位名稱` in('條件','條件');

6.篩選資料表欄位為sales開頭的資料

SELECT * FROM `資料表名稱` WHERE left(`欄位名稱`,5)='sales' ORDER BY `欄位名稱`;

7.統計篩算後欄位個數

SELECT count(*) FROM `資料表名稱` WHERE left(`欄位名稱`,5)='sales' ORDER BY `欄位名稱`;

8.使用like指定做模糊比對

SELECT * FROM `資料表名稱` WHERE `欄位名稱` like 'sales%';

9.使用範圍設定來顯示特定文字

SELECT `欄位名稱` CASE when `欄位名稱`>=100 then '滿分' when `欄位名稱`>=60 then '及格' else '不及格' END as '別名' FROM `資料表名稱`
;

10.合併應用

SELECT `欄位名稱` as '別名',sum(`欄位名稱`) as '別名',`欄位名稱` CASE when sum(`欄位名稱`)>=100 then '滿分' when sum(uria)>=60 then '及格' else '不及格' END as '別名' FROM `資料表名稱` GROUP BY `欄位名稱`
;

11.HAVING篩選

SELECT `欄位名稱` as '別名',sum(`欄位名稱`) as '別名',`欄位名稱` CASE when sum(`欄位名稱`)>=100 then '滿分' when sum(uria)>=60 then '及格' else '不及格' END as '別名' FROM `資料表名稱` GROUP BY `欄位名稱` HAVING sum(`欄位名稱`)>=100
;
