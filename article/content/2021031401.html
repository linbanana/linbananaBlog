SQL指令應用(查詢1)


1.查詢資料表顯示所有欄位

select * from `資料表名稱`;

2.查詢資料表顯示特別欄位加入別名

SELECT `欄位名稱` as '別名' FROM `資料表名稱`;

3.排除重覆值

SELECT DISTINCT `欄位名稱` FROM `資料表名稱`;

4.顯示目前排序後之設定筆數

SELECT * FROM `資料表名稱` LIMIT 50;

5.使用欄位排序

SELECT * FROM `資料表名稱` ORDER BY `欄位名稱`;

6.自訂兩欄位以上排序

SELECT * FROM `資料表名稱` ORDER by `欄位名稱` DESC,`欄位名稱` ASC;

7.顯示指定特定欄位篩選值

SELECT * FROM `資料表名稱` WHERE `欄位名稱`='條件';

8.篩選與排序複合

SELECT * FROM `資料表名稱` WHERE `欄位名稱`='條件' ORDER by `欄位名稱` DESC;

9.加入關係運算子條件篩選

SELECT * FROM `資料表名稱` WHERE `欄位名稱`='條件' or `欄位名稱`='條件';
